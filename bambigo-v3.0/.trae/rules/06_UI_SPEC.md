# BambiGO 介面規格
# UI/UX 設計規範

---

## 🎯 本文件的使用方式

```
給 AI 開發代理的指引：

1. 採用「底部四分頁」單頁應用架構
2. 四個 Tab 對應 L1-L4：DNA / LIVE / FACILITY / BAMBI
3. L1 DNA 有三層結構（主類別→次類別→景點），有導航按鈕
4. L3 FACILITY 只有兩層（設施類別→位置），無導航按鈕
5. L4 BAMBI 是完整的行程規劃助手
6. 頂部共用區域顯示節點名稱、天氣、地圖預覽
```

```typescript
const SUPPORTED_LOCALES = ['zh-TW', 'ja', 'en'] as const;
type Locale = typeof SUPPORTED_LOCALES[number];

// 語系偵測優先順序
// 1. URL 參數 (?lang=ja)
// 2. localStorage
// 3. 瀏覽器語系
// 4. 預設 zh-TW

function detectLocale(): Locale {
  // 1. URL 參數
  const urlParams = new URLSearchParams(window.location.search);
  const urlLang = urlParams.get('lang');
  if (urlLang && SUPPORTED_LOCALES.includes(urlLang as Locale)) {
    return urlLang as Locale;
  }
  
  // 2. localStorage
  const storedLang = localStorage.getItem('locale');
  if (storedLang && SUPPORTED_LOCALES.includes(storedLang as Locale)) {
    return storedLang as Locale;
  }
  
  // 3. 瀏覽器語系
  const browserLang = navigator.language.split('-')[0];
  if (browserLang === 'ja') return 'ja';
  if (browserLang === 'en') return 'en';
  if (browserLang === 'zh') return 'zh-TW';
  
  // 4. 預設
  return 'zh-TW';
}
```
